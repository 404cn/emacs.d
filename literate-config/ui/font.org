* set fonts

#+begin_src elisp
  (defun nasy/set-font ()
    "Nasy set font"
    (interactive)
    (set-face-attribute
     'default nil
     :font (font-spec :name   *font*
                      :weight *font-weight*
                      :size   *font-size*))

    (when (display-graphic-p)
      (dolist (charset '(kana han symbol cjk-misc bopomofo))
        (set-fontset-font
         (frame-parameter nil 'font)
         charset
         (font-spec :name   *font-cjk*
                    :weight *font-weight-cjk*
                    :size   *font-size-cjk*)))))

  (nasy/set-font)
#+end_src
