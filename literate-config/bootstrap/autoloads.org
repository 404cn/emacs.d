  (defun nasy/lsp-format-region-or-buffer ()
    "Format the buffer (or selection) with LSP."
    (interactive)
    (unless (bound-and-true-p lsp-mode)
      (user-error "Not in an LSP buffer"))
    (call-interactively
     (if (nasy/region-active-p)
         #'lsp-format-region
       #'lsp-format-buffer)))
