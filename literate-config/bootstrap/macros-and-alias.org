Useful Macros and Alias.

#+begin_src elisp
  (defmacro λ! (&rest body)
    "A shortcut for inline interactive lambdas."
    (declare (doc-string 1))
    `(lambda () (interactive) ,@body))

  (defalias 'lambda! 'λ!)
#+end_src
